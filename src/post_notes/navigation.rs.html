31impl From<RawTagNode> for TagNode {
32    fn from(raw_tag_node: RawTagNode) -> Self {
33        let mut child_tags = raw_tag_node
34            .child_tags
35            .into_iter()
36            .map(|value| value.1.into())
37            .collect::<Vec<TagNode>>();
38        child_tags.sort_unstable();
39        let mut files = raw_tag_node
40            .files
41            .into_iter()
42            .collect::<Vec<InternalLink>>();
43        files.sort_unstable();
44        Self {
45            tag: raw_tag_node.tag,
46            child_tags,
47            files,
48        }
49    }
50}

59#[derive(Clone, Debug, Serialize, Deserialize)]
60pub struct Navigation {
61    pub root: TagNode,
62}
63
64impl From<&Vec<PostNote>> for Navigation {
65    fn from(notes: &Vec<PostNote>) -> Self {
66        let mut root = RawTagNode::default();
67
68        for note in notes {
69            for tag in &note.properties.tags {
70                let parts: Vec<&str> = tag.split('/').filter(|p| !p.is_empty()).collect();
71
72                if parts.is_empty() {
73                    continue;
74                }
75
76                let mut current_node = &mut root;
77
78                for part in &parts {
79                    let tag_part = Tag::from(*part);
80
81                    current_node = current_node
82                        .child_tags
83                        .entry(tag_part.clone())
84                        .or_insert_with(|| RawTagNode {
85                            tag: tag_part,
86                            ..Default::default()
87                        });
88                }
89
90                current_node.files.insert(note.file_name.clone());
91
92                log::info!("Inserted {} under the tag {}", &*note.file_name, &**tag);
93            }
94        }
95
96        Navigation { root: root.into() }
97    }
98}
