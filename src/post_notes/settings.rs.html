87/// Read Settings from `Config.toml` or command line arguments.
88fn merge_settings(
89    default: Config,
90    file: Option<Config>,
91    args: Option<Config>,
92) -> Result<Settings, Error> {
93    let mut raw_settings = Config::builder().add_source(default);
94    if let Some(file) = file {
95        raw_settings = raw_settings.add_source(file);
96    }
97    if let Some(args) = args {
98        raw_settings = raw_settings.add_source(args);
99    };
100    Ok(raw_settings.build()?.try_deserialize::<Settings>()?)
101}

103/// Loads the configured settings from either `Config.toml` or the command line
104/// arguments.
105/// - If both are set the command line arguments overwrites the settings from
106///   the `Config.toml`.
107/// - If neither are set the default settings are used.
108pub fn get_settings() -> Settings {
109    let args = Args::parse();
110    // Interpret default settings.
111    let config_default = Config::try_from(&Settings::default())
112        .map_err(|err| log::error!("Could not interpret the default settings as config: {err}"))
113        .ok();
114    // Load and interpret config file.
115    let config_file = Config::builder()
116        .add_source(File::with_name(&args.config).required(false))
117        .build()
118        .map_err(|err| log::error!("Could not interpret config file: {err}"))
119        .ok();
120    // Interpret cli arguments.
121    let config_args = Config::try_from(&args)
122        .map_err(|err| log::error!("Could not interpret cli arguments: {err}"))
123        .ok();
124    // If we have a default config, try to merge everything.
125    if let Some(default) = config_default {
126        if let Ok(settings) = merge_settings(default, config_file, config_args) {
127            return settings;
128        }
129        log::error!("Could not merge settings.");
130    }
131    log::info!(
132        "Could not load settings from config file or command line arguments, using default settings instead."
133    );
134    Settings::default()
135}
